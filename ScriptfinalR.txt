#naive bayes
library(rsample)
set.seed(123)
split <- initial_split(corona, prop = 0.6, strata = "rezultat")
train <- training(split)
test <- testing(split)

table(train$ rezultat)
table(test$ rezultat)

features <- setdiff(names(train), "rezultat")
x <- train[,features]
y <- train$rezultat

train_control <- trainControl(
  method = "cv",
  number = 10 )


mod_nb1 <- train(
  x = x,
  y = y,
  method = "nb",
  trControl = train_control )
mod_nb1
confusionMatrix(mod_nb1)



search_grid <- expand.grid(
  usekernel = c(TRUE, FALSE),
  fL = 0.5,
  adjust = seq(0, 5, by = 1) )

mod_nb2 = train(
  x = x,
  y = y,
  method = "nb",
  trControl = train_control,
  tuneGrid = search_grid,
)
mod_nb2
confusionMatrix(mod_nb2)

mod_nb2$results %>%
  top_n(5, wt = Accuracy) %>%
  arrange(desc(Accuracy))



pred <- predict(mod_nb2, test)
pred
predProb <- predict(mod_nb2, test, type = "prob")
predProb
confusionMatrix(pred, test$rezultat)

library(pROC)

dataset <- data.frame(
  actual.class <- test$rezultat,
  probability <- predProb[,1]
)
roc.val <- roc(actual.class ~ probability, dataset)
roc.val
roc.val$specificities
roc.val$sensitivities
adf <- data.frame(specificity <- roc.val$specificities, sensitivity <- roc.val$sensitivities)

ggplot(adf, aes(specificity, sensitivity)) + geom_line(color = "red") + scale_x_reverse() + theme(text = element_text(size = 20))
